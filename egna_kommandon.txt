---  TEMPLATE-PIPELINE (extern research → dossiers → artifacts)  ---

npm run template-pipeline:refresh          # Full: scrape + import + hydrate + build + embeddings + validate
npm run template-pipeline:refresh:reuse-cache   # Samma men hoppa over scrape (snabbare)

---  ARTEFAKTER — återbygg (template-library + scaffold-research + v0-mallar m.m.)  ---

# Rekommenderat efter du har giltig scrape-summary under data/external-template-pipeline/:
npm run artifacts:rebuild                  # Snabb: återanvänder scrape-cache/repo-cache, purge + rebuild + validate + typecheck
npm run artifacts:rebuild:full             # Långsammare: kör även --refresh-scrape (ny scrape innan rebuild)
npm run artifacts:rebuild:with-eval        # Som rebuild men kör även eval efteråt
npm run artifacts:rebuild:full:with-eval # Full scrape + eval + typecheck

# Avancerat (samma script som npm anropar):
python scripts/rebuild_artifacts.py --help
python scripts/rebuild_artifacts.py --with-typecheck              # minimalt
python scripts/rebuild_artifacts.py --refresh-scrape --reset-repo-cache --with-typecheck  # hård reset av cache

# Efter lyckad rebuild kan du sätta SAJTMASKIN_STRICT_GENERATED_ARTIFACTS=true (eller ta bort =false)
# så att appen kräver att template-library.generated.json / scaffold-research.generated.json finns.

---  TEMPLATE-LIBRARY (enskilda steg)  ---

npm run template-library:import            # Importera discovery-data
npm run template-library:hydrate-cache     # Klona/uppdatera repo-cache (shallow)
npm run template-library:build             # Bygg template-library.generated.json + dossiers + scaffold-research
npm run template-library:validate-runtime  # Validera att catalog, embeddings och scaffold-research ar synkade
npm run template-library:verify-summary    # Verifiera discovered summary
npm run template-library:embeddings        # Generera template-library embeddings (OpenAI)

---  SCAFFOLDS  ---

npm run scaffolds:status                   # Status over alla scaffolds
npm run scaffolds:validate                 # Validera manifest-filer
npm run scaffolds:embeddings               # Generera scaffold-embeddings.json (OpenAI)
npm run scaffolds:eval                     # Kor scaffold-selection eval
npm run scaffolds:curate                   # Curate scaffold candidates fran template-library
npm run scaffolds:all                      # Kor alla steg i ordning
python scripts/scaffolds/scaffold_cli.py status       # Python CLI: status
python scripts/scaffolds/scaffold_cli.py all --typecheck  # Python CLI: allt + typecheck

---  SCAFFOLD-VARIANTER  ---

npx tsx scripts/scaffolds/derive-variants-from-dossiers.ts          # @deprecated — gammal pipeline
npm run scaffolds:variant-embeddings                                # Embeddings for alla 21 variants (OpenAI)

---  DOSSIERS (legoklossar: integrationer + UI-sektioner)  ---

# Status / overview:
npm run dossiers:index                     # Bygg om master.json + by-category.json (snabbt)
npm run dossiers:embeddings                # Regenerera dossier-embeddings.json (OpenAI, ~30 sek)
npm run dossiers:rebuild                   # index + recommendations:merge + embeddings i sekvens

# AI-kurering (GPT-5.4 gor grovjobbet, $1-3 per batch):
npm run dossiers:curate                    # Alla drafts (~12 min)
npm run dossiers:curate -- --only=<id>     # En specifik draft
npm run dossiers:curate -- --dry-run --only=<id>   # Forhandsgranska utan att skriva
npm run dossiers:curate -- --limit=5       # Max 5 at gangen
npm run dossiers:curate -- --force --only=<id>     # Re-kurera redan aktiv

# Smoke-test (verifiera att LLM-flodet plockar ratt dossiers):
npx tsx scripts/dossiers/smoke-select.ts   # 5 realistiska prompts -> selected dossiers

# Pipeline-steg (sallan):
npm run dossiers:scrape                    # Playwright: hamta katalog (~3 min, 419 templates)
npm run dossiers:enrich                    # Playwright: hamta detaljer (~14 min)
npm run dossiers:import                    # Importera enriched -> skiss-filer
npm run dossiers:queue                     # Bygg _curation-queue.md
npm run dossiers:promote                   # Promota fran curated-promotions.txt -> drafts
npm run dossiers:clone-repos               # Shallow-klona alla draft-repon (~5 min, 290 MB)
npm run dossiers:extract-files             # Extrahera filer + deps + env fran cache
npm run dossiers:recommend:merge           # Regenerera scaffold-recommendations (bevara dina edits)
npm run dossiers:recommend:force           # Skriv over scaffold-recommendations helt
