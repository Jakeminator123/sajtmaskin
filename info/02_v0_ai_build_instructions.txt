-------------------------------------------------
```typescript
import { generateText } from 'ai';
import { createVercel } from '@ai-sdk/vercel';

// Create Vercel provider with API key
const vercel = createVercel({
  apiKey: process.env.V0_API_KEY,
});

// Three quality levels for users
const MODEL_MAP = {
  budget: 'v0-1.5-md',    // âš¡ Fast, cheap ($1.5/$7.5 per 1M tokens)
  standard: 'v0-1.5-md',  // â­ Balanced (same model as budget)
  premium: 'v0-1.5-lg',   // ğŸ’ Best quality ($15/$75 per 1M tokens)
} as const;

type QualityLevel = keyof typeof MODEL_MAP;

export async function generateCode(
  prompt: string, 
  quality: QualityLevel = 'standard'
) {
  const modelId = MODEL_MAP[quality];
  
  const { text, usage } = await generateText({
    model: vercel(modelId),
    prompt,
    maxTokens: quality === 'premium' ? 4000 : 2000,
  });
  
  return {
    code: text,
    model: modelId,
    tokens: usage,
  };
}

// For chat-based iteration
export async function refineCode(
  existingCode: string,
  instruction: string,
  quality: QualityLevel = 'standard'
) {
  const prompt = `
Here is the existing code:
\`\`\`
${existingCode}
\`\`\`

Please modify it according to this instruction: ${instruction}

Return the complete updated code.
`;
  
  return generateCode(prompt, quality);
}
```


FILE: src/app/api/generate/route.ts
-----------------------------------
```typescript
import { generateCode } from '@/lib/v0-generator';
import { NextRequest, NextResponse } from 'next/server';

export async function POST(req: NextRequest) {
  try {
    const { prompt, quality = 'standard' } = await req.json();
    
    if (!prompt || prompt.trim().length === 0) {
      return NextResponse.json(
        { success: false, error: 'Prompt is required' },
        { status: 400 }
      );
    }
    
    const result = await generateCode(prompt, quality);
    
    // Sanitize response - remove any v0/Vercel references
    const cleanedCode = result.code
      .replace(/v0\.dev/gi, '')
      .replace(/v0\.app/gi, '')
      .replace(/vercel/gi, '');
    
    return NextResponse.json({
      success: true,
      code: cleanedCode,
      model: result.model,
      tokens: result.tokens,
    });
  } catch (error) {
    console.error('Generation error:', error);
    return NextResponse.json(
      { success: false, error: 'Generation failed. Please try again.' },
      { status: 500 }
    );
  }
}
```

export async function POST(req: NextRequest) {

}

// Frontend calls YOUR API, not v0 directly

type QualityLevel = 'budget' | 'standard' | 'premium';

export async function generateWebsite(
  prompt: string, 
  quality: QualityLevel = 'standard'
) {
  const res = await fetch('/api/generate', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ prompt, quality }),
  });
  
  if (!res.ok) {
    throw new Error('Generation failed');
  }
  
  return res.json();
}

export async function refineWebsite(
  existingCode: string,
  instruction: string,
  quality: QualityLevel = 'standard'
) {
  const res = await fetch('/api/refine', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ existingCode, instruction, quality }),
  });
  
  if (!res.ok) {
    throw new Error('Refinement failed');
  }
  
  return res.json();
}
