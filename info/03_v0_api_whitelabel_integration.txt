```typescript
// src/lib/v0-generator.ts

import { generateText } from 'ai';
import { createVercel } from '@ai-sdk/vercel';

const vercel = createVercel({
  apiKey: process.env.V0_API_KEY,
});

// Map user-facing options to v0 models
const MODEL_MAP = {
  standard: 'v0-1.5-md',  // Standard quality, good for most cases
  premium: 'v0-1.5-lg',   // Best quality, 10x cost
} as const;

type QualityLevel = keyof typeof MODEL_MAP;

export async function generateCode(
  prompt: string, 
  quality: QualityLevel = 'standard'
) {
  const modelId = MODEL_MAP[quality];
  
  const { text, usage } = await generateText({
    model: vercel(modelId),
    prompt,
    maxTokens: quality === 'premium' ? 4000 : 2000,
  });
  
  return {
    code: text,
    model: modelId,
    tokens: usage,
  };
}
```

File: src/app/api/generate/route.ts

```typescript
import { NextRequest, NextResponse } from 'next/server';

const V0_API_KEY = process.env.V0_API_KEY;
const V0_API_BASE = process.env.V0_API_BASE || 'https://api.v0.dev';

export async function POST(req: NextRequest) {
  try {
    const { prompt, chatId } = await req.json();

    // Call v0 API from backend (hidden from user)
    const v0Response = await fetch(`${V0_API_BASE}/v1/chat/completions`, {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${V0_API_KEY}`,
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({
        model: 'v0-1.5-md',
        messages: [{ role: 'user', content: prompt }],
        stream: false,
      }),
    });

    const data = await v0Response.json();

    // IMPORTANT: Filter the response to remove any v0 references
    const cleanedResponse = sanitizeResponse(data);

    return NextResponse.json(cleanedResponse);
  } catch (error) {
    // Generic error - don't expose v0 details
    return NextResponse.json(
      { error: 'Generation failed. Please try again.' },
      { status: 500 }
    );
  }
}

// Remove any v0 branding from response
function sanitizeResponse(data: any) {
  const stringified = JSON.stringify(data);
  
  // Replace v0 references with your brand
  const cleaned = stringified
    .replace(/v0\.dev/gi, 'sajtmaskin.se')
    .replace(/v0\.app/gi, 'sajtmaskin.se')
    .replace(/vercel/gi, '')
    .replace(/v0/gi, 'SajtMaskin AI');
  
  return JSON.parse(cleaned);
}
```

STEP 3: Platform API Wrapper (for chat-based generation)

File: src/lib/ai-generator.ts

```typescript
// This is YOUR wrapper - users never see v0

interface GenerationResult {
  id: string;
  code: string;
  files: Array<{ name: string; content: string }>;
  previewHtml?: string;
}

export class AIGenerator {
  private baseUrl = '/api';  // Your backend, not v0

  async createProject(description: string): Promise<GenerationResult> {
    const res = await fetch(`${this.baseUrl}/generate`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ prompt: description }),
    });

    if (!res.ok) {
      throw new Error('Generation failed');
    }

    return res.json();
  }

  async refineProject(projectId: string, instruction: string): Promise<GenerationResult> {
    const res = await fetch(`${this.baseUrl}/generate`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ 
        chatId: projectId, 
        prompt: instruction 
      }),
    });

    return res.json();
  }

  async getProjectFiles(projectId: string): Promise<Array<{ name: string; content: string }>> {
    const res = await fetch(`${this.baseUrl}/project/${projectId}/files`);
    return res.json();
  }
}

// Export singleton
export const aiGenerator = new AIGenerator();
```

```typescript
// src/app/api/preview/[projectId]/route.ts

export async function GET(
  req: NextRequest,
  { params }: { params: { projectId: string } }
) {
  // Fetch the generated files from your database/cache
  const files = await getProjectFiles(params.projectId);
  
  // Find the main page file
  const mainFile = files.find(f => f.name.includes('page.tsx'));
  
  if (!mainFile) {
    return new Response('Preview not available', { status: 404 });
  }

  // Build a standalone HTML preview
  const previewHtml = buildPreviewHtml(files);
  
  return new Response(previewHtml, {
    headers: { 'Content-Type': 'text/html' },
  });
}

function buildPreviewHtml(files: Array<{ name: string; content: string }>): string {
  // Combine all CSS and create a static preview
  const styles = files
    .filter(f => f.name.endsWith('.css'))
    .map(f => f.content)
    .join('\n');

  return `
    <!DOCTYPE html>
    <html>
    <head>
      <meta charset="utf-8">
      <meta name="viewport" content="width=device-width, initial-scale=1">
      <title>Preview - SajtMaskin</title>

      <style>${styles}</style>
    </head>
    <body>
      <div id="preview-root">
        <!-- Rendered components go here -->

  `;
}
```

================================================================================
                    CACHING FOR COST REDUCTION & SPEED
================================================================================

v0 API costs tokens. Cache responses to reduce costs and hide latency.

File: src/lib/cache.ts

```typescript
import { Redis } from '@upstash/redis';  // Or use in-memory for dev

const redis = new Redis({
  url: process.env.REDIS_URL,
  token: process.env.REDIS_TOKEN,
});

export async function getCachedGeneration(promptHash: string) {
  return redis.get(`gen:${promptHash}`);
}

export async function cacheGeneration(promptHash: string, result: any) {
  // Cache for 24 hours
  await redis.setex(`gen:${promptHash}`, 86400, JSON.stringify(result));
}

export function hashPrompt(prompt: string): string {
  // Simple hash for cache key
  return Buffer.from(prompt).toString('base64').slice(0, 32);
}
```

Usage in API route:
```typescript
const promptHash = hashPrompt(prompt);
const cached = await getCachedGeneration(promptHash);

if (cached) {
  return NextResponse.json(cached);  // Fast response, no v0 call
}

// Otherwise call v0 and cache result
const result = await callV0Api(prompt);
await cacheGeneration(promptHash, result);
return NextResponse.json(result);
```

================================================================================
                    STREAMING RESPONSES (Optional)
================================================================================

For better UX, stream the generation progress to the user.
This hides the fact that v0 is being called in background.

File: src/app/api/generate/stream/route.ts

```typescript
export async function POST(req: NextRequest) {
  const { prompt } = await req.json();

  const encoder = new TextEncoder();
  const stream = new ReadableStream({
    async start(controller) {
      // Send progress updates while waiting for v0
      controller.enqueue(encoder.encode('data: {"status": "analyzing"}\n\n'));
      
      await sleep(1000);
      controller.enqueue(encoder.encode('data: {"status": "designing"}\n\n'));
      
      await sleep(1000);
      controller.enqueue(encoder.encode('data: {"status": "generating"}\n\n'));

      // Actually call v0 (user doesn't see this)
      const result = await callV0Api(prompt);

      // Send final result
      controller.enqueue(encoder.encode(`data: {"status": "complete", "result": ${JSON.stringify(result)}}\n\n`));
      controller.close();
    },
  });

  return new Response(stream, {
    headers: {
      'Content-Type': 'text/event-stream',
      'Cache-Control': 'no-cache',
      'Connection': 'keep-alive',
    },
  });
}
```
