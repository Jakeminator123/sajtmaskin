function readUiComponent(name: string, searchDirs: string[]): string | null {
  const filename = `${name}.tsx`;
  for (const dir of searchDirs) {
    const fullPath = nodePath.join(dir, filename);
    try {
      return fs.readFileSync(fullPath, "utf-8");
    } catch {
      continue;
    }
  }
  return null;
}
